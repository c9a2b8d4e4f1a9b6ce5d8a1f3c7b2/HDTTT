### Title
Any Authenticated User Can Corrupt Transaction Group Structure via Missing Ownership Check in `removeTransactionGroup`

### Summary
`removeTransactionGroup` in `TransactionGroupsService` fetches a group and all its `TransactionGroupItem` records with no user-ownership filter. It then deletes each `TransactionGroupItem` row **before** the per-transaction ownership check in `removeTransaction` can fire. When `removeTransaction` throws `UnauthorizedException`, the `TransactionGroupItem` row is already permanently deleted, leaving the group in a corrupted, partially-dismembered state. Any verified, authenticated user can trigger this against any group ID.

  async removeTransactionGroup(user: User, id: number): Promise<boolean> {
    const group = await this.dataSource.manager.findOneBy(TransactionGroup, { id });
    if (!group) {
      throw new Error('group not found');
    }
    const groupItems = await this.dataSource.manager.find(TransactionGroupItem, {
      relations: {
        group: true,
      },
      where: {
        group: {
          id: group.id,
        },
      },
    });
    for (const groupItem of groupItems) {
      const transactionId = groupItem.transactionId;
      await this.dataSource.manager.remove(TransactionGroupItem, groupItem);
      await this.transactionsService.removeTransaction(transactionId, user, false);
    }

    await this.dataSource.manager.remove(TransactionGroup, group);

    emitTransactionUpdate(this.notificationsPublisher, groupItems.map(gi => ({ entityId: gi.transactionId })));

    return true;
  }
